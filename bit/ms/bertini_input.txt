CONFIG
TRACKTYPE: 1;
INPUT
	function f1, f2, f3;
	variable_group M, F_N, F_C;
	constant V_s, V_m, V_d, k_s, K, K_m, K_d, k_in, k_out, n;
	
	V_s = 4;
	V_m = 1.5;
	V_d = 1.0;
	k_s = 0.5;
	K = 0.2;
	K_m = 0.15;
	K_d = 0.15;
	k_in = 0.0248;
	k_out = 0.1;
	n = 4;
	
	f1 = V_s * K^n / (K^n + F_N^n) - V_m * M / (K_m + M);
	f2 = k_s * M - V_d * F_C / (K_d + F_C) - k_in * F_C + k_out * F_N;
	f3 = k_in * F_C - k_out * F_N;
END;
